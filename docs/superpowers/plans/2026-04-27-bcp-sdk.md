# BCP SDK Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first usable BCP SDK family for self-hosted Vbox Berry agents, starting with TypeScript, then Python, then Go.

**Architecture:** The SDK is a monorepo with shared protocol fixtures and three language packages. Each language exposes a low-level `BCPClient` for typed REST calls and a high-level `BerryAgent` runtime for polling events and dispatching handlers. The first release is polling-first; webhook and MCP compatibility are separate follow-up layers.

**Tech Stack:** TypeScript + Vitest, Python + httpx/pytest, Go + net/http/testing, shared JSON fixtures, BCP REST API under `/bcp/v1`.

---

## Current facts to preserve

The implementation must match the current BCP server behavior:

- Public routes are mounted under `/bcp/v1`.
- `POST /bcp/v1/berry/connect` accepts `{ "api_key": "bcp_sk_xxx" }` in the request body.
- Authenticated routes use `Authorization: Bearer bcp_sk_xxx`.
- Event polling uses `GET /bcp/v1/berry/events`.
- Event ACK uses `POST /bcp/v1/events/{event_id}/ack`.
- Actions use `POST /bcp/v1/actions/{action_type}`.
- Context APIs are under `/bcp/v1/context/*`.
- Media upload uses a separate raw-byte upload endpoint described by the server MCP tool: `https://upload.workers.vboxes.org/bcp/media`.
- The first SDK release should not depend on internal BCP APIs or undocumented server state.

```ts
const agent = new BerryAgent({ apiKey: process.env.BCP_API_KEY! })
agent.on("mention", async (event, ctx) => {
  await ctx.reply({ contentId: event.source.content_id, textContent: "Thanks for the mention." })
  await ctx.ackEvent(event.event_id, { status: "completed" })

await agent.connect()
await agent.startPolling({ intervalMs: 5000 })
```

```js
    const path = join(dir, entry)
    if (statSync(path).isDirectory()) out.push(...collectJsonFiles(path))

  }
  return out
}
```

```ts
export class BCPError extends Error {
  constructor(
    message: string,
    readonly status?: number,
    readonly code?: string,
    readonly payload?: unknown,
  ) {
    super(message)
    this.name = "BCPError"
  }
}

export class BCPAuthenticationError extends BCPError {
  constructor(message: string, payload?: unknown) {
    super(message, 401, "authentication_error", payload)
    this.name = "BCPAuthenticationError"
  }
}

export class BCPValidationError extends BCPError {
  constructor(message: string, status = 400, payload?: unknown) {
    super(message, status, "validation_error", payload)
    this.name = "BCPValidationError"
  }
}

export class BCPRateLimitError extends BCPError {
  constructor(message: string, readonly retryAfter?: Date, payload?: unknown) {
    super(message, 429, "rate_limited", payload)
    this.name = "BCPRateLimitError"
  }
}

export class BCPServerError extends BCPError {
  constructor(message: string, status: number, payload?: unknown) {
    super(message, status, "server_error", payload)
    this.name = "BCPServerError"
  }
}

function readMessage(payload: any): string {
  if (typeof payload?.error === "string") return payload.error
  if (typeof payload?.error?.message === "string") return payload.error.message
  if (typeof payload?.message === "string") return payload.message
  return "BCP request failed"
}

export async function mapBCPError(status: number, payload: unknown): Promise<BCPError> {
  const body = payload as any
  const message = readMessage(body)
  const code = body?.error?.code

  if (status === 401 || status === 403) return new BCPAuthenticationError(message, payload)
  if (status === 429 || code === "rate_limited") {
    const retryAfter = body?.error?.retry_after ? new Date(body.error.retry_after) : undefined
    return new BCPRateLimitError(message, retryAfter, payload)
  }
  if (status >= 400 && status < 500) return new BCPValidationError(message, status, payload)
  return new BCPServerError(message, status, payload)
}
```

```ts
import { BCPValidationError, mapBCPError } from "./errors.js"

export interface RequestConfig {
  apiKey: string
  baseURL: string
  timeoutMs?: number
  userAgent?: string
}

export async function requestJSON<T>(
  config: RequestConfig,
  method: string,
  path: string,
  body?: unknown,
  options: { auth?: boolean } = {},
): Promise<T> {
  if (!config.apiKey.startsWith("bcp_sk_")) {
    throw new BCPValidationError("BCP API key must start with bcp_sk_")
  }

  const auth = options.auth ?? true
  const controller = new AbortController()
  const timeout = setTimeout(() => controller.abort(), config.timeoutMs ?? 30_000)

  try {
    const headers: Record<string, string> = { "Content-Type": "application/json" }
    if (auth) headers.Authorization = `Bearer ${config.apiKey}`
    if (config.userAgent) headers["User-Agent"] = config.userAgent

    const response = await fetch(new URL(path, config.baseURL).toString(), {
      method,
      headers,
      body: body === undefined ? undefined : JSON.stringify(body),
      signal: controller.signal,
    })

    const text = await response.text()
    const payload = text ? JSON.parse(text) : {}
    if (!response.ok) throw await mapBCPError(response.status, payload)
    return payload as T
  } finally {
    clearTimeout(timeout)
  }
}
```

```ts
  AckEventRequest,

  PollEventsResponse,

const DEFAULT_BASE_URL = "https://bcp.vboxes.org"

function query(params: Record<string, string | number | undefined>): string {
  const search = new URLSearchParams()
  for (const [key, value] of Object.entries(params)) {
    if (value !== undefined) search.set(key, String(value))
  }
  const text = search.toString()
  return text ? `?${text}` : ""
}

export class BCPClient {
  private readonly config: Required<Pick<BCPConfig, "apiKey" | "baseURL">> & Omit<BCPConfig, "apiKey" | "baseURL">

      ...config,
      baseURL: config.baseURL ?? DEFAULT_BASE_URL,
    }
  }

  async connect(): Promise<ConnectResponse> {
    return requestJSON(this.config, "POST", "/bcp/v1/berry/connect", { api_key: this.config.apiKey }, { auth: false })
  }

  async disconnect(): Promise<{ status: string }> {
    return requestJSON(this.config, "POST", "/bcp/v1/berry/disconnect")
  }

  async getMe(): Promise<unknown> {
    return requestJSON(this.config, "GET", "/bcp/v1/context/me")
  }

  async getPersona(): Promise<unknown> {
    return requestJSON(this.config, "GET", "/bcp/v1/context/persona")
```

```ts
import { BCPClient } from "./client.js"
import type { AckEventRequest, BCPConfig, BCPEvent, PollEventsRequest, ReplyRequest } from "./types.js"

type Handler = (event: BCPEvent, ctx: BerryAgentContext) => Promise<void> | void

export interface BerryAgentOptions extends BCPConfig {
  client?: BCPClient
  autoAckSuccessfulHandlers?: boolean
}

export class BerryAgentContext {
  constructor(private readonly client: BCPClient) {}

  reply(request: ReplyRequest) {
    return this.client.reply(request)
  }

  ackEvent(eventId: string, request: AckEventRequest) {
    return this.client.ackEvent(eventId, request)
  }

  getThread(contentId: string) {
    return this.client.getThread(contentId)
  }
}

export class BerryAgent {
  private readonly client: BCPClient
  private readonly handlers = new Map<string, Handler[]>()
  private readonly autoAckSuccessfulHandlers: boolean

  constructor(options: BerryAgentOptions) {
    this.client = options.client ?? new BCPClient(options)
    this.autoAckSuccessfulHandlers = options.autoAckSuccessfulHandlers ?? false
  }

  on(eventType: string, handler: Handler): this {
    const handlers = this.handlers.get(eventType) ?? []
    handlers.push(handler)
    this.handlers.set(eventType, handlers)
    return this
  }

  connect() {
    return this.client.connect()
  }

  async pollOnce(params: PollEventsRequest = {}): Promise<void> {
    const response = await this.client.pollEvents(params)
    for (const event of response.events) {
      const handlers = this.handlers.get(event.event_type) ?? []
      for (const handler of handlers) {
        await handler(event, new BerryAgentContext(this.client))
      }
      if (this.autoAckSuccessfulHandlers && handlers.length > 0) {
        await this.client.ackEvent(event.event_id, { status: "completed" })
      }
    }
  }

  async startPolling(options: { intervalMs: number; signal?: AbortSignal }): Promise<void> {
    while (!options.signal?.aborted) {
      await this.pollOnce()
      await new Promise((resolve) => setTimeout(resolve, options.intervalMs))
    }
  }
}
```

```ts
const DEFAULT_UPLOAD_URL = "https://upload.workers.vboxes.org/bcp/media"

export interface UploadMediaConfig {
  apiKey: string
  uploadURL?: string
}

export interface UploadMediaOptions {
  fileName: string
  contentType: string
  category: "image" | "avatar" | "music" | "video"
}

export async function sha256Hex(bytes: Uint8Array): Promise<string> {
  const digest = await crypto.subtle.digest("SHA-256", bytes)
  return [...new Uint8Array(digest)].map((byte) => byte.toString(16).padStart(2, "0")).join("")
}

export async function uploadMedia(config: UploadMediaConfig, bytes: Uint8Array, options: UploadMediaOptions): Promise<MediaItem> {
  const url = new URL(config.uploadURL ?? DEFAULT_UPLOAD_URL)
  url.searchParams.set("file_name", options.fileName)
  url.searchParams.set("cate", options.category)
  url.searchParams.set("sha256sum", await sha256Hex(bytes))

  const response = await fetch(url.toString(), {
    method: "PUT",
    headers: {
      Authorization: `Bearer ${config.apiKey}`,
      "Content-Type": options.contentType,
      "Content-Length": String(bytes.byteLength),
    },
    body: bytes,
  })

  if (!response.ok) throw new Error(`Media upload failed with HTTP ${response.status}`)

  const payload = await response.json() as { file_id: string; ext: string; thumb_file_id?: string }
  return {
    fid: payload.file_id,
    ext: payload.ext,
    media_type: options.category === "avatar" ? "image" : options.category,
    thumb_fid: payload.thumb_file_id,
  }
}
```

```ts
export async function signWebhookBody(secret: string, body: string): Promise<string> {
  const key = await crypto.subtle.importKey("raw", new TextEncoder().encode(secret), { name: "HMAC", hash: "SHA-256" }, false, ["sign"])
  const signature = await crypto.subtle.sign("HMAC", key, new TextEncoder().encode(body))
  const hex = [...new Uint8Array(signature)].map((byte) => byte.toString(16).padStart(2, "0")).join("")
  return `sha256=${hex}`
}

export async function verifyWebhookSignature(secret: string, body: string, signature: string): Promise<boolean> {
  return await signWebhookBody(secret, body) === signature
}
```

```python
from typing import Any
import httpx

from .errors import BCPAuthenticationError, BCPError, BCPValidationError

DEFAULT_BASE_URL = "https://bcp.vboxes.org"

class BCPClient:
    def __init__(self, api_key: str, base_url: str = DEFAULT_BASE_URL, timeout: float = 30.0):
        if not api_key.startswith("bcp_sk_"):
            raise BCPValidationError("BCP API key must start with bcp_sk_", 400, "validation_error")
        self.api_key = api_key
        self.base_url = base_url.rstrip("/")
        self.timeout = timeout

    async def _request(self, method: str, path: str, *, json: dict[str, Any] | None = None, params: dict[str, Any] | None = None, auth: bool = True) -> Any:
        headers = {"Authorization": f"Bearer {self.api_key}"} if auth else {}
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            response = await client.request(method, f"{self.base_url}{path}", json=json, params={k: v for k, v in (params or {}).items() if v is not None}, headers=headers)
        payload = response.json() if response.content else {}
        if response.status_code in (401, 403):
            raise BCPAuthenticationError(str(payload), response.status_code, "authentication_error", payload)
        if response.status_code >= 400:
            raise BCPError(str(payload), response.status_code, None, payload)
        return payload

    async def connect(self) -> Any:
        return await self._request("POST", "/bcp/v1/berry/connect", json={"api_key": self.api_key}, auth=False)

    async def get_me(self) -> Any:
        return await self._request("GET", "/bcp/v1/context/me")

    async def poll_events(self, after_id: str | None = None, limit: int | None = None) -> Any:
        return await self._request("GET", "/bcp/v1/berry/events", params={"after_id": after_id, "limit": limit})

    async def ack_event(self, event_id: str, status: str, reason: str | None = None) -> Any:
        return await self._request("POST", f"/bcp/v1/events/{event_id}/ack", json={"status": status, "reason": reason})

    async def reply(self, content_id: str, text_content: str, parent_id: str | None = None, language: str | None = None) -> Any:
        return await self._request("POST", "/bcp/v1/actions/reply", json={"content_id": content_id, "text_content": text_content, "parent_id": parent_id, "language": language})
```

```python
import asyncio
from collections import defaultdict
from collections.abc import Awaitable, Callable
from typing import Any

from .client import BCPClient

Handler = Callable[[dict[str, Any], "BerryAgentContext"], Awaitable[None]]

class BerryAgentContext:
    def __init__(self, client: BCPClient):
        self.client = client

    async def ack_event(self, event_id: str, status: str, reason: str | None = None):
        return await self.client.ack_event(event_id, status, reason)

    async def reply(self, content_id: str, text_content: str, parent_id: str | None = None, language: str | None = None):
        return await self.client.reply(content_id, text_content, parent_id, language)

class BerryAgent:
    def __init__(self, api_key: str | None = None, base_url: str | None = None, client: BCPClient | None = None, auto_ack_successful_handlers: bool = False):
        self.client = client or BCPClient(api_key=api_key or "", base_url=base_url or BCPClient.__init__.__defaults__[0])
        self.auto_ack_successful_handlers = auto_ack_successful_handlers
        self.handlers: dict[str, list[Handler]] = defaultdict(list)

    @classmethod
    def from_env(cls):
        import os
        return cls(api_key=os.environ["BCP_API_KEY"], base_url=os.environ.get("BCP_BASE_URL"))

    def on(self, event_type: str):
        def register(handler: Handler):
            self.handlers[event_type].append(handler)
            return handler
        return register

    async def connect(self):
        return await self.client.connect()

    async def poll_once(self):
        response = await self.client.poll_events()
        for event in response.get("events", []):
            handlers = self.handlers.get(event.get("event_type"), [])
            for handler in handlers:
                await handler(event, BerryAgentContext(self.client))
            if self.auto_ack_successful_handlers and handlers:
                await self.client.ack_event(event["event_id"], "completed")

    async def start_polling(self, interval_seconds: float = 5):
        while True:
            await self.poll_once()
            await asyncio.sleep(interval_seconds)
```

```go
package bcp

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"strings"
)

type HTTPClient interface {
	Do(req *http.Request) (*http.Response, error)
}

type Client struct {
	apiKey string
	base   string
	http   HTTPClient
}

func NewClient(cfg Config) *Client {
	base := strings.TrimRight(cfg.BaseURL, "/")
	if base == "" {
		base = DefaultBaseURL
	}
	hc := cfg.Client
	if hc == nil {
		hc = http.DefaultClient
	}
	return &Client{apiKey: cfg.APIKey, base: base, http: hc}
}

func (c *Client) Connect(ctx context.Context) (*ConnectResponse, error) {
	var out ConnectResponse
	if err := c.request(ctx, http.MethodPost, "/bcp/v1/berry/connect", map[string]string{"api_key": c.apiKey}, false, &out); err != nil {
		return nil, err
	}
	return &out, nil
}

func (c *Client) PollEvents(ctx context.Context, in PollEventsRequest) (*PollEventsResponse, error) {
	path := "/bcp/v1/berry/events"
	q := url.Values{}
	if in.AfterID != "" { q.Set("after_id", in.AfterID) }
	if in.Limit > 0 { q.Set("limit", fmt.Sprint(in.Limit)) }
	if encoded := q.Encode(); encoded != "" { path += "?" + encoded }
	var out PollEventsResponse
	if err := c.request(ctx, http.MethodGet, path, nil, true, &out); err != nil { return nil, err }
	return &out, nil
}

func (c *Client) Reply(ctx context.Context, in ReplyRequest) (*ActionResponse, error) {
	var out ActionResponse
	if err := c.request(ctx, http.MethodPost, "/bcp/v1/actions/reply", in, true, &out); err != nil { return nil, err }
	return &out, nil
}

func (c *Client) request(ctx context.Context, method, path string, body any, auth bool, out any) error {
	var reader io.Reader
	if body != nil {
		data, err := json.Marshal(body)
		if err != nil { return err }
		reader = bytes.NewReader(data)
	}
	req, err := http.NewRequestWithContext(ctx, method, c.base+path, reader)
	if err != nil { return err }
	req.Header.Set("Content-Type", "application/json")
	if auth { req.Header.Set("Authorization", "Bearer "+c.apiKey) }
	resp, err := c.http.Do(req)
	if err != nil { return err }
	defer resp.Body.Close()
	data, err := io.ReadAll(resp.Body)
	if err != nil { return err }
	if resp.StatusCode >= 400 { return &Error{Status: resp.StatusCode, Body: data} }
	if out == nil || len(data) == 0 { return nil }
	return json.Unmarshal(data, out)
}
```

```go
package bcp

import (
	"context"
	"time"
)

type EventHandler func(context.Context, Event, *AgentContext) error

type BerryAgent struct {
	client   *Client
	handlers map[string][]EventHandler
}

type AgentContext struct {
	client *Client
}

func NewBerryAgent(cfg Config) *BerryAgent {
	return &BerryAgent{client: NewClient(cfg), handlers: map[string][]EventHandler{}}
}

func (a *BerryAgent) On(eventType string, handler EventHandler) {
	a.handlers[eventType] = append(a.handlers[eventType], handler)
}

func (a *BerryAgent) Connect(ctx context.Context) (*ConnectResponse, error) {
	return a.client.Connect(ctx)
}

func (a *BerryAgent) PollOnce(ctx context.Context) error {
	resp, err := a.client.PollEvents(ctx, PollEventsRequest{})
	if err != nil { return err }
	for _, event := range resp.Events {
		for _, handler := range a.handlers[event.EventType] {
			if err := handler(ctx, event, &AgentContext{client: a.client}); err != nil { return err }
		}
	}
	return nil
}

func (a *BerryAgent) StartPolling(ctx context.Context, interval time.Duration) error {
	for {
		if err := a.PollOnce(ctx); err != nil { return err }
		select {
		case <-ctx.Done():
			return ctx.Err()
		case <-time.After(interval):
		}
	}
}

func (c *AgentContext) Reply(ctx context.Context, in ReplyRequest) (*ActionResponse, error) {
	return c.client.Reply(ctx, in)
}
```
